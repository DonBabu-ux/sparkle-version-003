```
    if not user.email_verified: return { error: "Verify email first" }

function createSession(user):
    payload = {
        userId: user.id,
        email: user.email,
        role: user.role,
        exp: now + (rememberMe ? 30 days : 24 hours)
    }
    token = jwt.sign(payload, SECRET_KEY)
    setCookie("auth_token", token, {
        httpOnly: true,
        secure: true,
        sameSite: "strict",
        maxAge: rememberMe ? 2592000 : 86400
    })
    // Also store session in database
    db.createSession({
        userId: user.id,
        token: token,
        ipAddress: request.ip,
        userAgent: request.userAgent,

1.9 RATE LIMIT ATTEMPTS
ALGORITHM:
function rateLimit(email, ip):
    key = `login_attempts:${email}:${ip}`
    attempts = redis.get(key) or 0
    if attempts >= 5:
        ttl = redis.ttl(key)
        if ttl > 0:
            return { blocked: true, remainingSeconds: ttl }
    redis.incr(key)
    if attempts == 0:
        redis.expire(key, 900) // 15 minutes
    return { blocked: false }

1.10 REMEMBER ME
ALGORITHM:
- Checkbox state: rememberMe = true/false
- Pass to createSession function
- If true: JWT expiry = 30 days, cookie maxAge = 30 days
- If false: JWT expiry = 24 hours, cookie maxAge = 24 hours
```

### 👁️ ALGORITHM 31: PROFILE VISITOR

```
31.1-31.18 WHAT VISITOR SEES
ALGORITHM:
function getProfileViewData(profileUserId, viewerUserId):
    profile = db.query("SELECT * FROM users WHERE id = $1", [profileUserId])
    
    // Always visible
    result = {
        name: profile.first_name + " " + profile.last_name,
        username: profile.username,
        avatar: profile.avatar_url,
        bio: profile.bio,
        followerCount: profile.followers_count,
        followingCount: profile.following_count,
        postCount: profile.posts_count,
        joinDate: profile.created_at,
        campus: profile.campus_id
    }
    
    // Check privacy
    if profile.is_private:
        isFollowing = db.query("SELECT 1 FROM follows WHERE follower_id = $1 AND following_id = $2 AND status = 'accepted'", 
                              [viewerUserId, profileUserId])
        if not isFollowing and profileUserId != viewerUserId:
            result.posts = null
            result.privateMessage = "This account is private"
            return result
    
    // Get posts based on visibility
    if profileUserId == viewerUserId:
        posts = db.query("SELECT * FROM posts WHERE user_id = $1 ORDER BY created_at DESC", [profileUserId])
    else:
        posts = db.query("""
            SELECT * FROM posts 
            WHERE user_id = $1 AND visibility IN ('public', 'friends') 
            ORDER BY created_at DESC
        """, [profileUserId])
    
    result.posts = posts
    result.canFollow = (profileUserId != viewerUserId AND not isFollowing)
    result.canMessage = (profileUserId != viewerUserId)
    
    // Conditional visibility
    result.email = profile.show_email ? profile.email : null
    result.phone = profile.show_phone ? profile.phone : null
    
    // Mutual followers
    result.mutualFollowers = db.query("""
        SELECT f1.follower_id 
        FROM follows f1
        JOIN follows f2 ON f1.follower_id = f2.following_id
        WHERE f1.following_id = $1 AND f2.follower_id = $2
    """, [profileUserId, viewerUserId])
    
    return result
```

```
function createNotification(eventType, data):
    // Determine recipient
    switch(eventType):
        case 'like':
            recipientId = data.post.user_id
            content = `${data.user.name} liked your post`
            type = 'like'
            break
        case 'comment':
            recipientId = data.post.user_id
            content = `${data.user.name} commented: ${data.comment.content}`
            type = 'comment'
            break
        case 'follow':
            recipientId = data.targetUserId
            content = `${data.user.name} started following you`
            type = 'follow'
            break
        case 'follow_request':
            recipientId = data.targetUserId
            content = `${data.user.name} wants to follow you`
            type = 'follow_request'
            break
        case 'message':
            recipientId = data.recipientId
```

```
function authMiddleware(req, res, next):
    token = req.cookies.auth_token
    if not token:
        return res.status(401).json({ error: "No token" })
    
    // Check blacklist
    isBlacklisted = redis.get(`blacklist:${token}`)
    if isBlacklisted:
        return res.status(401).json({ error: "Session revoked" })
    
    try:
        decoded = jwt.verify(token, SECRET_KEY)
    catch jwt.TokenExpiredError:
        return res.status(401).json({ error: "Token expired", code: "EXPIRED" })
    catch:
        return res.status(401).json({ error: "Invalid token" })
    
    req.userId = decoded.userId
    req.userRole = decoded.role
    next()

function refreshToken(req, res):
    oldToken = req.cookies.auth_token
    decoded = jwt.decode(oldToken)
    
    // Check if token is within refresh window
    if decoded.exp - Date.now()/1000 > 86400: // More than 24 hours left
        return res.status(400).json({ error: "Token not eligible for refresh" })
    
    // Create new token
    newPayload = {
        userId: decoded.userId,
        email: decoded.email,
        role: decoded.role,
        exp: Date.now()/1000 + (decoded.rememberMe ? 2592000 : 86400)
    }
    newToken = jwt.sign(newPayload, SECRET_KEY)
    
    // Update session in database
    db.query("""
        UPDATE user_sessions 
        SET token = $1, expires_at = $2, last_active_at = NOW()
        WHERE user_id = $3 AND token = $4
    """, [newToken, new Date(newPayload.exp * 1000), decoded.userId, oldToken])
    
    // Blacklist old token
    redis.setex(`blacklist:${oldToken}`, 86400, 'refreshed')
    
    // Set new cookie
    res.cookie('auth_token', newToken, {
        httpOnly: true,
        secure: true,
        sameSite: 'strict',
        maxAge: decoded.rememberMe ? 2592000000 : 86400000
    })
    
    return res.json({ success: true })
```

```
function confirm2FA(userId, verificationCode):

            token: token,
```
